function verifierCSRFToken($token) {
    return isset($_SESSION['csrf_token']) && hash_equals($_SESSION['csrf_token'], $token);
}

if (!verifierCSRFToken($_POST['csrf_token'])) {
    die('Erreur : Token CSRF invalide.');
}
